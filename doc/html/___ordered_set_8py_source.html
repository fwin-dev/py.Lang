12 ##
13 # A set that remembers the order elements were added.
14 #
15 # The internal self.__map dictionary maps keys to links in a doubly linked list.
16 # The circular doubly linked list starts and ends with a sentinel element.
17 # The sentinel element never gets deleted (this simplifies the algorithm).
18 # The prev/next links are weakref proxies (to prevent circular references). Individual links are kept alive by
19 # the hard reference in self.__map. Those hard references disappear when a key is deleted from an OrderedSet.
20 #
21 class OrderedSet(collections.MutableSet):

30  def __setitem__(self, index, value):
31  if isinstance(index, slice):

69  def _iterLinks(self, index):
70  if isinstance(index, int):
71  index = self._getPositiveIndex(index)
72  yield self._getLink_byIndex(index)
73  elif isinstance(index, slice):
74  index = self._convertSlice(index)
75  nextLink = self._getLink_byIndex(index.start)
76  betweenStep = 0 # if betweenStep, don't yield; 0 == False, > 0 == True
77  iterAll_step = max(1, min(-1, index.step)) # -1 if negative step, +1 if positive step
78  iterAll_stop = index.stop - index.start + iterAll_step # + iterAll_step because range stops 1 short of last number
79  for _ in range(0, iterAll_stop, iterAll_step):
80  if betweenStep == 0:
81  yield nextLink
82  betweenStep %= abs(index.step)
83  if index.step > 0: nextLink = nextLink.next
84  else: nextLink = nextLink.prev
85  else:

109  def _getLink_byIndex(self, index):
110  index = self._getPositiveIndex(index)
111  root = self.__root
112  if abs(index) < len(self) / 2: # faster to iterate in forward order
113  curr = root.next
114  for i in range(0, index + 1):
115  if i == index:
116  return curr
117  curr = curr.next

135  #
136  # Inserts newElemBefore into the current position of oldLinkAfter. oldLinkAfter will then come after newElemBefore.
137  #
138  # oldLinkAfter --- Must be an existing link in the map.
139  # newElemBefore --- New element to insert.
140  # updateOnExist --- If True and newElemBefore is already in the list, the already existing newElemBefore will be
141  # removed and the newElemBefore passed into this function will be inserted before oldLinkAfter.
142  # If False and newElemBefore is already in the list, newElemBefore will not be inserted, and
143  # the whole list will not be modified.
144  #
145  # Returns True if link was inserted, False if not inserted because it already exists and updateOnExist is False.
146  #
147  def _insertBefore_link(self, newElemBefore, oldLinkAfter, updateOnExist):
148  if oldLinkAfter != self.__root and oldLinkAfter.key not in self.__map:
149  raise ValueError(str(oldLinkAfter.key) + " is not in OrderedSet")
150  if newElemBefore in self.__map:
151  if updateOnExist:
152  self.discard(newElemBefore)
153  else:
154  return False
155 
156  self.__map[newElemBefore] = link = Link(newElemBefore)
157  link.prev, link.next = oldLinkAfter.prev, oldLinkAfter
158  link.prev.next = oldLinkAfter.prev = proxy(link)
159  return True
